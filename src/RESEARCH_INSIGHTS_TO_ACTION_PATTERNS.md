```typescript
function navigateToGoalsWithFilter(filters: {
  status?: 'completed' | 'at-risk' | 'on-track';
  category?: string;
  timeline?: 'month' | 'quarter' | 'year';
}) {
  // Step 1: Switch to Goals tab view
  setActiveGoalView('list');
  
  // Step 2: Apply filters with animation
  if (filters.status) {
    setActiveGoalStatusFilter(filters.status);
  }
  if (filters.category) {
    setActiveGoalCategoryFilter(filters.category);
  }
  
  // Step 3: Scroll to goals list
  setTimeout(() => {
    document.getElementById('goals-list')?.scrollIntoView({ 
      behavior: 'smooth',
      block: 'start'
    });
  }, 300);
  
  // Step 4: Show toast confirmation
  toast.success(`Showing ${filters.status} goals`, {
    description: 'Clear filter to see all goals',
    action: {
      label: 'Clear',
      onClick: () => setActiveGoalStatusFilter('all')
    }
  });
  
  // Step 5: Track analytics
  trackInsightAction('navigate_to_filtered_goals', filters);
}
```
